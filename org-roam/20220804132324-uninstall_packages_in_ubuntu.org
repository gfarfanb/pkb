#+title: Uninstall packages in Ubuntu

Clear all
#+begin_src
  sudo apt-get purge --auto-remove <package>
#+end_src

Set default
#+begin_src
  sudo update-alternatives --config <package>
#+end_src

Restart service
#+begin_src
    sudo service <service> restart
#+end_src
