#+title: Uninstall packages in Ubuntu
#+filetags: :maintenance:ubuntu:

#+begin_src
  sudo apt-get purge --auto-remove packagename # clear all

  sudo update-alternatives --config packagename # set default

  sudo service servicename restart # restart service
#+end_src
