#+title: Install Emacs on Windows WSL (Debian distro) - gfarfanb setup

- [ ] Start [[id:1e047c08-2dcb-4ac5-8ce3-fe3c7f4edfa0][Windows Terminal]] and open a new tab with *Debian* profile
  - Create new Unix username
- [ ] Update the package sources list
  #+begin_src sh
    sudo apt update
  #+end_src
- [ ] Install Emacs
  #+begin_src sh
    sudo apt install emacs
    emacs --version
  #+end_src
- [ ] Clone *pkb* repo
  #+begin_src sh
    mkdir workspace
    cd workspace
    git clone https://github.com/gfarfanb/pkb.git
    cd pkb
    git checkout -b org-roam
    git pull origin org-roam
  #+end_src
- [ ] Create new Emacs initialization files
  #+begin_src
    C-x C-f <REF> ~/.emacs <RET>
    C-x C-s
    C-x C-f <REF> ~/.emacs.d/custom-vars.el <RET>
    C-x C-s
  #+end_src
- [ ] Create a Lisp directory ~\~/.emacs.d/lisp~ for file packages
  #+begin_src
    C-x C-f <RET> ~/.emacs.d/ <RET> (+) lisp <RET>
  #+end_src
